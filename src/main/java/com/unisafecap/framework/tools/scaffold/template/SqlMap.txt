<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE mapper PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN" "http://mybatis.org/dtd/mybatis-3-mapper.dtd" >

<mapper namespace="${modelPath}">

	<resultMap id="result" type="${modelPath}">
${resultMap}
	</resultMap>

	<sql id="base_field">
		${baseField}
	</sql>
	
	<sql id="other-condition">
${otherCondition}
 	</sql>
 	
 	<sql id="like-condition">
${likeCondition}
 	</sql>
	
	<sql id="selectByMapSQL">
		select
		<include refid="base_field" />
		from ${tblName}
		<choose>
		    <when test="findBy != null">
		      	<where>
					<include refid="other-condition" />
				</where>
		    </when>
		    <when test="findByLike != null">
		      	<where>
					<include refid="like-condition" />
				</where>
		    </when>
		</choose>
		<if test="sortName != null and sortName != '' and sortName != 'null'">
			order by ${sortName} ${sortBy}
		</if>
	</sql>
	
	<select id="selectByMap" resultMap="result" >
		<include refid="selectByMapSQL" />
	</select>
	
	<select id="selectCount" resultType="int" >
		select count(*) from ${tblName} 
		<choose>
		    <when test="findBy != null">
		      	<where>
					<include refid="other-condition" />
				</where>
		    </when>
		    <when test="findByLike != null">
		      	<where>
					<include refid="like-condition" />
				</where>
		    </when>
		</choose>
	</select>
	
	<select id="selectPage" resultMap="result" >
		<include refid="selectByMapSQL" />
		 limit #{startRow},#{pageSize}
	</select>
	
	<select id="select" resultMap="result">
		select
		<include refid="base_field" />
		from ${tblName}  
		<where>
			<if test="_parameter != null">
				id = #{id,jdbcType=BIGINT}
			</if>
		</where>
	</select>
	
	<select id="selectByIds" resultMap="result" >
		select
		<include refid="base_field" />
		from ${tblName}  where id in (${idlist})
	</select>		

	<insert id="insert" parameterType="${modelPath}" useGeneratedKeys="true" keyProperty="id">
		${insertStatement}
	</insert>
	
	<insert id="insert4Selective" parameterType="${modelPath}" useGeneratedKeys="true" keyProperty="id">
		${insertMapModel}
	</insert>

	<update id="update" parameterType="${modelPath}">
		${updateStatement}
	</update>	
	
	<update id="update4Selective" parameterType="${modelPath}">
		${updateMapModel}
	</update>

	<delete id="deleteByPrimaryKey">
		delete from ${tblName} where id = #{id,jdbcType=BIGINT}
	</delete>
    
	<delete id="delete" parameterType="${modelPath}">
		delete from ${tblName}
		<where>
			<if test="id != null">
				id = #{id,jdbcType=BIGINT}
			</if>
			<include refid="other-condition" />
		</where>
	</delete>	
		
</mapper>
